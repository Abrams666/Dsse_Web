CREATE TABLE news (
  id SERIAL PRIMARY KEY,
  title VARCHAR(255) NOT NULL,
  tag SMALLINT NOT NULL,
  date DATE NOT NULL,
  publishDate DATE NOT NULL DEFAULT CURRENT_DATE,
  content VARCHAR(1024),
  fileLinks VARCHAR(255)[]
)
